   function uniform(xl,xu)

      real(wp),intent(in) :: xl !! lower bound
      real(wp),intent(in) :: xu !! upper bound
      real(wp) :: uniform

      uniform = xl + (xu-xl)*uniform_random_number()

   end function uniform
